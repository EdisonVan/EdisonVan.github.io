根据 moduleId 返回模块级权限树

Authorized.ts

export function fetchPermissionButton(moduleId: string) {
  return http.get(`${urls.LOGINAPI}/Auth/UserInfo`, {
    moduleId: moduleId,
    systemCode: 'DIYCODE', // 对应系统的Code码
  });
}

/* 按钮权限，不符合权限不显示 */
interface IAuthorized {
  src: string[] // 权限数组
  moduleId: string // 权限对应的 moduleId
}

export default function Authorized(props: IAuthorized) {
  const { src, moduleId, children } = props
  return src.some((item) => item.moduleId === moduleId) ? children : ''
}

Useage Code Snippet

const [permissionButton, setPermissionButton] = useState(null);
useEffect(() => {
  const moduleId = 'YHYWXT_P_00005'; // 页面权限Code
  // 预先加载页面按钮权限
  fetchPermissionButton(moduleId)
    .then((res) => {
      const { isSuccess, result, message } = res.data;
      if (isSuccess) {
        setPermissionButton(result.privilegeTree);
      } else {
        message.error(message);
      }
    })
    .catch((err) => {
      console.log(err);
    });
}, []);
return (
  permissionButton && (
    <Authorized src={permissionButton} moduleId='YHYWXT_B_00010'>
      <Button size='small'>添加</Button>
    </Authorized>
  )
);
